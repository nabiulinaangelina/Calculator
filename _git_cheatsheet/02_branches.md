# Ветки
```bash
# Показать список веток
git branch

# Создать новую ветку
git branch <name>

# Задаёт основную ветку
# git branch — работа с ветками.
# -M — это флаг, который означает "переименовать ветку принудительно" (move/rename).
# main — новое имя ветки.
# Таким образом, команда переименовывает текущую ветку в main и делает её основной (главной) веткой проекта.
# По умолчанию Git раньше создавал ветку master.
# Сейчас в большинстве проектов (например, на GitHub) используется название main как основная ветка.
# Если ветка с таким именем уже существует, флаг -M принудительно заменит её (в отличие от -m, который выдаст ошибку при конфликте имён).
git branch -M main

# Переключиться на ветку
git checkout <name>

# Создать и сразу перейти в новую ветку
git checkout -b <name>

# Объединить изменения из другой ветки.
git merge <branch>
```
